        localStorage.setItem('points', points);
        updateLevel();
        
        if (points >= level * 100) {
            showPopup(`🎉 LEVEL UP! You're now Level ${level + 1}!`, "🌟");
            createConfetti();
        }
    }
    
    function checkStreak() {
        const today = new Date().toDateString();
        if (lastClickDate === today) return;
        
        const yesterday = new Date();
        yesterday.setDate(yesterday.getDate() - 1);
        
        if (lastClickDate === yesterday.toDateString()) {
            streak++;
            if (streak === 3) {
                showPopup("🔥 3-Day Streak! +50 XP!", "🔥");
                addPoints(50);
            } else if (streak === 7) {
                showPopup("🏆 7-Day Streak! +100 XP!", "🏆");
                addPoints(100);
            }
        } else if (lastClickDate !== today) {
            streak = 1;
        }
        
        localStorage.setItem('streak', streak);
        localStorage.setItem('lastClickDate', today);
        document.getElementById('streak').innerText = streak;
    }
    
    function showPopup(message, emoji) {
        const popup = document.createElement('div');
        popup.className = 'achievement-popup';
        popup.innerHTML = `${emoji} ${message}`;
        document.body.appendChild(popup);
        setTimeout(() => popup.remove(), 2500);
    }

            confetti.style.pointerEvents = 'none';
            confetti.style.zIndex = '9999';
            confetti.style.animation = `fall ${Math.random() * 2 + 1}s linear forwards`;
            document.body.appendChild(confetti);
            setTimeout(() => confetti.remove(), 2000);
        }
    }

        else if (currentFilter === "cashback") filtered = allOffers.filter(o => o.type === "cashback");
        else if (currentFilter === "high_value") filtered = allOffers.filter(o => o.reward >= 150);
        return filtered;
    }

        const container = document.getElementById('offersContainer');
        const filtered = getFilteredOffers();
        
        if (filtered.length === 0) {
